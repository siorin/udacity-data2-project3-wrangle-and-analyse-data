# read in the cleaned archive
df = pd.read_csv('twitter_archive_master.csv')

# reconvert timestamp column to datetime
df.timestamp = pd.to_datetime(df.timestamp)

# add 3 columns for all 3 predictions 
# - if prediction 1 is a dog type, use p1 and p1_conf
# - else if prediction 2 is a dog type, use p2 and p2_conf
# - else if prediction 3 is a dog type, use p3 and p3_conf

df['predict'] = df.p1
df['predict_percent'] = df.p1_conf
df['predict_dog'] = df.p1_dog

first_guess = df[df.p1_dog == True]
print("guess " + str(1) + " : " + str(sum(df.predict_dog == True)))

# first guess not a dog type, but second guess is a dog type
second_guess = df[(df['p1_dog']==False) & (df['p2_dog']==True)]

# first and second guesses not a dog type, but third guess is a dog type
third_guess = df[(df['p1_dog']==False) & (df['p2_dog']==False) & (df['p3_dog']==True)]

missing_guesses = []

for i in second_guess.index:
    try:
        j = df[df.tweet_id == second_guess.loc[i].tweet_id].index[0]
        df.loc[j, 'predict'] = second_guess.loc[i].p2
        df.loc[j, 'predict_percent'] = second_guess.loc[i].p2_conf
        df.loc[j, 'predict_dog'] = True
    except:
        # these were removed during cleaning when dropping retweets and duplicate expanded_urls
        missing_guesses.append(second_guess.loc[i].tweet_id)
# print how many prediction values are dog types
print("guess " + str(2) + " : " + str(sum(df.predict_dog == True)))

for i in third_guess.index:
    try:
        j = df[df.tweet_id == third_guess.loc[i].tweet_id].index[0]
        df.loc[j, 'predict'] = third_guess.loc[i].p3
        df.loc[j, 'predict_percent'] = third_guess.loc[i].p3_conf
        df.loc[j, 'predict_dog'] = True
    except:
        # these were removed during cleaning when dropping retweets and duplicate expanded_urls
        missing_guesses.append(third_guess.loc[i].tweet_id)
# print how many prediction values are dog types
print("guess " + str(3) + " : " + str(sum(df.predict_dog == True)))
